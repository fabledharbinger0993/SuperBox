# rekordbox-toolkit Python dependencies
#
# Install with: pip install -r requirements.txt
#
# System binaries required separately (see README):
#   macOS:   brew install ffmpeg chromaprint
#   Windows: winget install ffmpeg  (fpcalc: https://acoustid.org/chromaprint)
#
# pyrekordbox 0.4.4 is the tested version. The add_content, add_to_playlist,
# and update_content_path call signatures are version-sensitive — test after
# any pyrekordbox upgrade before distributing a new release.

flask>=3.0
flask-sock>=0.7
flask-limiter>=3.5
# Web framework — routes, SSE streaming, JSON responses, rate limiting

waitress>=3.0
# Production WSGI server (Automator-safe: no stdout bleed unlike dev server)

numpy>=1.26
# Required by librosa for audio analysis

pyrekordbox==0.4.4
# Pin any pyrekordbox upgrade before distributing a new release.

mutagen==1.47
# Audio tag reading and writing

librosa>=0.10.2
# BPM detection, beat tracking, key detection, and chroma analysis
pyloudnorm==0.1.1
# Loudness measurement (EBU R128 / LUFS)

soundfile>=0.12.1
# Audio file I/O used by pyloudnorm and normalisation verification

sounddevice>=0.4.6
# Audio playback used by the preview/monitoring feature in app.py

pyacoustid==1.3
# Acoustic fingerprinting wrapper; requires fpcalc binary (see above)

yt-dlp>=2024.1
# Audio download from Bandcamp, Beatport, Soundcloud, and direct URLs.
# Requires ffmpeg (already a system dep) for post-download format conversion.

qrcode[svg]>=7.4
# QR code generation for the FableGo pairing panel (inline SVG, no Pillow needed).

pywebview>=5.0
# Native OS window (WKWebView on macOS) — replaces the browser launch step.
# Enables real drag-and-drop path delivery and a proper .app experience.

PyQt5>=5.15
# Multimedia widgets for splash video playback on launch.
# Falls back gracefully if not available (skips splash and launches directly).

pyinstaller>=6.0
# Bundles Python + all deps into a standalone .app.  Run build.sh to produce
# dist/FableGear.app.  Not needed at runtime — dev/build tool only.
